read_mv_component( comp ) {
}
    @@mv_sign                                                           T
    @@mv_class                                                          S
    if ( mv_class == MV_CLASS_0 ) {
        @@mv_class0_bit                                                 T
        @@mv_class0_fr                                                  S
        @@mv_class0_hp                                                  T
        mag = ( ( mv_class0_bit << 3 ) |
                ( mv_class0_fr << 1 ) |
                  mv_class0_hp ) + 1
    } else {
        d = 0
        for ( i = 0; i < mv_class; i++ ) {
            @@mv_bit                                                    T
            d |= mv_bit << i
        }
        mag = CLASS0_SIZE << ( mv_class + 2 )
        @@mv_fr                                                         S
        @@mv_hp                                                         T
        mag += ( ( d << 3 ) | ( mv_fr << 1 ) | mv_hp ) + 1
    }
    return mv_sign ? -mag : mag
}
