#### 6.5.8 Scale mv function

<div class="syntax">
scale_mv( refList, refFrame ) {                                       <b>Type</b>
    candFrame = CandidateFrame[ refList ]
    if ( ref_frame_sign_bias[ candFrame ] != ref_frame_sign_bias[ refFrame ] )
        for ( j = 0; j < 2; j++ )
            CandidateMv[ refList ][ j ] *= -1
}
