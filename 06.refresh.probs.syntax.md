refresh_probs( ) {
    if ( error_resilient_mode == 0 ) {
        load_probs( frame_context_idx )
        adapt_coef_probs( )
        if ( FrameIsIntra == 0 ) {
            load_probs2( frame_context_idx )
            adapt_noncoef_probs( )
        }
        average_cdf( )
    }
    if ( refresh_frame_context )
        save_probs( frame_context_idx )
}
