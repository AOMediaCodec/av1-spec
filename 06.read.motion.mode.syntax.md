read_motion_mode( isCompound ) {
    if (MiSize < BLOCK_8x8) {
        motion_mode = SIMPLE
        return
    }
    if ( YMode == ZEROMV || YMode == ZERO_ZEROMV) {
        if (gm_type[ RefFrame[ 0 ] ] > TRANSLATION) {
            motion_mode = GLOBALWARP
            return
        }
    }
    if ( isCompound || !has_overlappable_candidates( ) ) {
        motion_mode = SIMPLE
        return
    }
    find_warp_samples()
    if ( NumSamples == 0) {
        @@use_obmc                                                           S
        motion_mode = use_obmc ? OBMC : SIMPLE
    } else {
        @@motion_mode                                                        S
    }
}
