#### Assign MV Function

<div class="syntax">
assign_mv( isCompound ) {
    Mv[ 1 ] = ZeroMv
    for ( i = 0; i < 1 + isCompound; i++ ) {
        if ( YMode == NEWMV )
            read_mv( i )
        else if ( YMode == NEARESTMV )
            Mv[ i ] = NearestMv[ i ]
        else if ( YMode == NEARMV )
            Mv[ i ] = NearMv[ i ]
        else
            Mv[ i ] = ZeroMv
    }
}
