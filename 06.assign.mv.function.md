#### Assign MV Function

<div class="syntax">
assign_mv( isCompound ) {
    Mv[ 1 ] = ZeroMv
    for ( i = 0; i < 1 + isCompound; i++ ) {
        if ( y_mode == NEWMV )
            read_mv( i )
        else if ( y_mode == NEARESTMV )
            Mv[ i ] = NearestMv[ i ]
        else if ( y_mode == NEARMV )
            Mv[ i ] = NearMv[ i ]
        else
            Mv[ i ] = ZeroMv
    }
}
