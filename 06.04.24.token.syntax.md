tokens( plane, startX, startY, txSz, blockIdx ) {                     <b>Type</b>
    segEob = 16 << (txSz << 1)
    scan = get_scan( plane, txSz, blockIdx )
    checkEob = 1
    for( c = 0; c < segEob; c++ ) {
        pos = scan[ c ]
        band = (txSz == TX_4X4) ? coefband_4x4[ c ] : coefband_8x8plus[ c ]
        if ( checkEob ) {
        <b>more_coefs</b>                                                    T
            if ( more_coefs == 0 )
                break
        }
        <b>token</b>                                                         T
        TokenCache[ pos ] = energyClass[ token ]
        if ( token == ZERO_TOKEN ) {
            Tokens[ pos ] = 0
            checkEob = 0
        } else {
            coef = read_coef( token )
            <b>sign_bit</b>                                                  L(1)
            Tokens[ pos ] = sign_bit ? -coef : coef
            checkEob = 1
        }
    }
    nonzero = c > 0
    EobTotal += nonzero
    for ( i = c; i < segEob; i++ )
        Tokens[ scan[ i ] ] = 0
    return nonzero
    }
}
