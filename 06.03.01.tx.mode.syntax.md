#### 6.3.1 TX mode syntax

<div class="syntax">
read_tx_mode( ) {                                                     <b>Type</b>
    if ( Lossless == 1 ) {
        TxMode = ONLY_4X4
    } else {
        <b>tx_mode_select</b>                                                L(1)
        if ( tx_mode_select == TX_MODE_SELECT ) {
            <b>TxMode = tx_mode_select</b>                                   L(1)
        } else {
            <b>tx_mode</b>                                                   L(2)
            TxMode = tx_mode
        }
    }
}

</div>
