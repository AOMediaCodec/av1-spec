#### Decode Tile Function

~~~~~
decode_tile( ) {
    if ( !dependent_tiles || TileNum == tg_start)
        clear_above_context( )
    DeltaLF = 0
    for ( r = MiRowStart; r < MiRowEnd; r += 16 ) {
        clear_left_context( )
        for ( c = MiColStart; c < MiColEnd; c += 16 ) {
            ReadDeltas = delta_q_present
            clear_block_decoded_flags( c < ( MiColEnd - 1 ) )
            decode_partition( r, c, BLOCK_64X64 )
        }
    }
}
