<div class="syntax">
decode_tile( ) {
    if ( !dependent_tiles || TileNum == tg_start)
        clear_above_context( )
    DeltaLF = 0
    for ( r = MiRowStart; r < MiRowEnd; r += 8 ) {
        clear_left_context( )
        for ( c = MiColStart; c < MiColEnd; c += 8 ) {
            ReadDeltas = delta_q_present
            decode_partition( r, c, BLOCK_64X64 )
        }
    }
}
